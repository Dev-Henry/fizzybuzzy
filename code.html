function getValues() {

    //get the user values from the page 
    let fizzValue = document.getElementById("fizzValue").value;
    let buzzValue = document.getElementById("buzzValue").value;

    //check to make sure they are numbers 
    fizzValue = parseInt(fizzValue);
    buzzValue = parseInt(buzzValue);

    // check to make sure they are integers 
    if (Number.isInteger(fizzValue) && Number.isInteger(buzzValue)) {

        //call fizzBuzz
        let fbArray = fizzBuzzC(fizzValue, buzzValue);

        //call displayData and write the values to the screen 
        displayData(fbArray);        
    }else {
        alert("You need to enter an integer")
    }
 }

//do fizz buzz
function fizzBuzz(fizzValue, buzzValue) {
    let returnArray = [];

    //loop form 1 to 100
    for (let i = 1; i <= 100; i++) {

         //check to see if it is divisible by both 3 and 5 
         //check to see if divisible by fizz value (3)
         //check to see if divisible by buzz value (5)
        if (i % fizzValue == 0 && i % buzzValue == 0) {
            returnArray.push('FizzBuzz');
        }else if (i % fizzValue == 0) {
            returnArray.push('Fizz');
        }else if (i % buzzValue == 0) {
            returnArray.push('Buzz');
        }else {
            returnArray.push(i);
        }
    }
    return returnArray;
}

function displayData(fbArray){

    //get the table body element from the page 
    let tableBody = document.getElementById("results");

    //get the template row 
    let templateRow = document.getElementById("fbTemplate");

    //clear the table first 
    tableBody.innerHTML = "";

    for (let index = 0; index < fbArray.length; index += 5) {

        let tableRow = document.importNode(templateRow.content, true);

        //grab the td to put into an array 

        let rowCols = tableRow.querySelectorAll("td");
        rowCols[0].classList.add(fbArray[index]);
        rowCols[0].textContent = fbArray[index];

        rowCols[1].classList.add(fbArray[index+1]);
        rowCols[1].textContent = fbArray[index+1];

        rowCols[2].classList.add(fbArray[index+2]);
        rowCols[2].textContent = fbArray[index+2];

        rowCols[3].classList.add(fbArray[index+3]);
        rowCols[3].textContent = fbArray[index+3];

        rowCols[4].classList.add(fbArray[index+4]);
        rowCols[4].textContent = fbArray[index+4];

        tableBody.appendChild(tableRow);
    }    
}

FizzyBuzzy Project Documentation
OverView

The FizzBuzz project is a web-based application that implements the classic FizzBuzz problem with customizable divisors. Users can input two integer values via a web form, and the program generates a sequence of numbers from 1 to 100, replacing multiples of Fizzy Value with "Fizz," multiples of Buzzy Value with "Buzz," and multiples of both with "FizzBuzz." The results are displayed in a table format on the webpage.

Purpose

This project demonstrates basic JavaScript programming concepts, including DOM manipulation, user input validation, and multiple approaches to solving the FizzBuzz problem. It serves as both an educational tool and a practical exercise in front-end development.

Features

1. Input: Users can input two integers to customize the Fizz and Buzz values.
2. FizzBuzz Calculation: The application computes the FizzBuzz output for numbers 1 to 100.
3. Dynamic Display: The results are displayed in a table format, updating in real-time as the user changes input values.
4. Error Handling: The app ensures that only integer inputs are accepted, providing feedback if the input is invalid.

The Breakdown of the Code (the third solution is the active solution in the synthax displayed)

This function retrieves the Fizz and Buzz values entered by the user. It ensures that the inputs are integers using parseInt(), then proceeds to call the fizzBuzzC() function to calculate the FizzBuzz output and display it using displayData()

fizzBuzz()

This is the core function that implements the classic FizzBuzz algorithm. It loops from 1 to 100 and checks if the current number is divisible by the Fizz and Buzz values. The result is added to an array, which is then returned.

fizzBuzz(B)

This is an alternative implementation of FizzBuzz using switch statements. It works similarly to fizzBuzz() but with a different syntax and structure. This is use of boolean variables with a switch case statement.

fizzBuzz(C)

This version uses a ternary operator to generate the FizzBuzz output more concisely, returning the result in an array.

displayData()

This function takes the array generated by the FizzBuzz functions and dynamically fills up an HTML table. It populates the table with the corresponding FizzBuzz results, updating the page's content accordingly and auto clears.
